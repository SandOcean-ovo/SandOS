31uint32_t *OS_StackInit(OS_TaskFunc_t task_function, void* task_param, uint32_t *stack_init_address, uint32_t stack_depth)
32{
33 extern void __global_pointer$;
34 /* 第一步：找到栈顶 */
35 uint32_t *sp = stack_init_address + stack_depth;
36
37 /* 第二步：字节对齐 */
38 sp = (uint32_t *)((uint32_t)sp & 0xFFFFFFF0); // 先把sp转成uint32_t，再把最后4位抹成0，最后转成uint32_t *
39
40 /* 第三步：填入数据 */
41 *(--sp) = (uint32_t)MSTATUS_VALUE; // mstatus 机器模式
42 *(--sp) = (uint32_t)task_function; // mepc
43 *(--sp) = (uint32_t)OS_TaskReturn; // x1：返回地址 类似LR
44 *(--sp) = (uint32_t)&__global_pointer$; // gp：全局指针，程序使用gp间接访问全局变量的值，这个值在程序编译完后就固定了，不能填0不然就飞了
45 *(--sp) = (uint32_t)task_param; // a0：函数参数
46 for(int i = 0; i < 27; ++i){
47 *(--sp) = (uint32_t)0x0;
48 }
49 /* 第四步：返回sp */
50 return sp;
51}
